# Grfico comparativo de caractersticas
caractersticas = ['danceability', 'energy', 'valence', 'acousticness', 'instrumentalness']
me_gusta_vals = [df[df['liked']==1][c].mean() for c in caractersticas]
no_me_gusta_vals = [df[df['liked']==0][c].mean() for c in caractersticas]

x = np.arange(len(caractersticas))
width = 0.35

fig, ax = plt.subplots(figsize=(10, 6))
bars1 = ax.bar(x - width/2, me_gusta_vals, width, label='Me gusta', color='#27ae60')
bars2 = ax.bar(x + width/2, no_me_gusta_vals, width, label='No me gusta', color='#e74c3c')

ax.set_xlabel('Caractersticas')
ax.set_ylabel('Promedio')
ax.set_title('Comparacin de Caractersticas Musicales')
ax.set_xticks(x)
ax.set_xticklabels(caractersticas, rotation=45)
ax.legend()

plt.tight_layout()
plt.show()
