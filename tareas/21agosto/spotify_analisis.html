# Gráfico comparativo de características
características = ['danceability', 'energy', 'valence', 'acousticness', 'instrumentalness']
me_gusta_vals = [df[df['liked']==1][c].mean() for c in características]
no_me_gusta_vals = [df[df['liked']==0][c].mean() for c in características]

x = np.arange(len(características))
width = 0.35

fig, ax = plt.subplots(figsize=(10, 6))
bars1 = ax.bar(x - width/2, me_gusta_vals, width, label='Me gusta', color='#27ae60')
bars2 = ax.bar(x + width/2, no_me_gusta_vals, width, label='No me gusta', color='#e74c3c')

ax.set_xlabel('Características')
ax.set_ylabel('Promedio')
ax.set_title('Comparación de Características Musicales')
ax.set_xticks(x)
ax.set_xticklabels(características, rotation=45)
ax.legend()

plt.tight_layout()
plt.show()
